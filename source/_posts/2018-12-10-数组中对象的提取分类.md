# 数据分类

```js
const list = [
  {
    label: 'a',
    value: 1
  },
  {
    label: 'c',
    value: 3
  },
  {
    label: 'b',
    value: 1
  },
  {
    label: 'a',
    value: 3
  },
  {
    label: 'c',
    value: 2
  }
]
const collect = {}
list.forEach(item => {
  // console.log(item)
  if (!collect[item.label]) {
    collect[item.label] = []
  }
  collect[item.label].push(item)
})
// console.log(collect)
collect = {
  a: [1, 3],
  b: [1],
  c: [3, 2]
}

// 可以使用entries(),keys(),values()来获取所需要的值
```
